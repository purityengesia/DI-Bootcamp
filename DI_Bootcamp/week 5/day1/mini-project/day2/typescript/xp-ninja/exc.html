Conditional Types, Keyof, and Type Safety

Exercise 1: Conditional Types

Using T extends U ? X : Y to map input types to output types.

// Define Conditional Type
type MappedType<T> = T extends number ? number : number;

function mapType<T extends number | string>(val: T): MappedType<T> {
    if (typeof val === "number") {
        return (val * val) as MappedType<T>;
    } else {
        return val.length as MappedType<T>;
    }
}

Exercise 2: Keyof & Lookup Types

Retrieving object properties dynamically while maintaining strict type safety.

function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

const user = { id: 1, name: "Gemini" };
const userName = getProperty(user, "name"); // Returns string

Exercise 3: Interfaces with Numeric Properties

Using index signatures to handle objects with dynamic keys but specific value types.

interface HasNumericProperty {
    [key: string]: number;
}

function multiplyProperty(obj: HasNumericProperty, key: string, factor: number): number {
    return obj[key] * factor;
}
