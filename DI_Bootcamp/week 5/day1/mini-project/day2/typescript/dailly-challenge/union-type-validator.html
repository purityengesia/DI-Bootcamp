Exercise: validateUnionType

This function performs Runtime Type Checking by comparing the typeof a value against an array of valid type strings.

function validateUnionType(value: any, allowedTypes: string[]): boolean {
    const currentType = typeof value;
    return allowedTypes.includes(currentType);
}

// Usage:
const isValid = validateUnionType("Hello", ["string", "number"]); // true
